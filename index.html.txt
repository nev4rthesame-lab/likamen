            if (!isCrashGameActive) return;
            
            crashMultiplier += 0.01;
            
            const multiplierElement = document.getElementById('crashMultiplier');
            multiplierElement.textContent = crashMultiplier.toFixed(2) + 'x';

            const progress = (crashMultiplier - 1) / (crashPoint - 1) * 100;
            chartLine.style.width = Math.min(progress, 100) + '%';
            
            if (crashMultiplier > crashPoint * 0.8) {
                multiplierElement.style.color = '#d63031';
                chartLine.style.background = '#d63031';
            }
            
            if (crashMultiplier >= crashPoint) {
                endCrashGame(false);
            }

            if (crashMultiplier > bestMultiplier) {

            updateBalanceDisplay();
            updateAllStats();
